## 二、Phase 3：崩溃处理

### 2.1 检测崩溃

```go
// 在 launch MC 之后，检查退出码和崩溃报告
func detectCrash(mcDir string, exitCode int) CrashInfo {
    crashInfo := CrashInfo{
        ExitCode: exitCode,
        Time:     time.Now(),
    }
    
    // 1. 退出码判断
    // Java 崩溃退出码通常不为 0
    // JVM crash: exit code 1 (uncaught exception)
    // OutOfMemory: exit code 1 或 3
    // 强制 kill: exit code -1
    if exitCode != 0 {
        crashInfo.IsCrash = true
        crashInfo.Reason = getExitCodeReason(exitCode)
    }
    
    // 2. 检查崩溃报告
    crashReportsDir := filepath.Join(mcDir, "crash-reports")
    if entries, err := os.ReadDir(crashReportsDir); err == nil {
        for _, e := range entries {
            if strings.HasSuffix(e.Name(), ".txt") {
                info, _ := e.Info()
                if time.Since(info.ModTime()) < 2*time.Minute {
                    // 2 分钟内的崩溃报告
                    crashInfo.IsCrash = true
                    crashInfo.HasCrashReport = true
                    crashInfo.CrashReportPath = filepath.Join(crashReportsDir, e.Name())
                    
                    // 读取崩溃原因
                    content, _ := os.ReadFile(crashInfo.CrashReportPath)
                    crashInfo.Reason = extractCrashReason(string(content))
                    break
                }
            }
        }
    }
    
    // 3. 检查 JVM 崩溃日志（HS_ERR）
    hsErrPattern := filepath.Join(mcDir, "hs_err_pid*.log")
    if matches, err := filepath.Glob(hsErrPattern); err == nil {
        for _, m := range matches {
            info, _ := os.Stat(m)
            if time.Since(info.ModTime()) < 2*time.Minute {
                crashInfo.IsCrash = true
                crashInfo.HasJVMError = true
                crashInfo.JVMPath = m
                break
            }
        }
    }
    
    return crashInfo
}

func getExitCodeReason(code int) string {
    switch code {
    case 0:
        return "正常退出"
    case 1:
        return "Java 未捕获异常"
    case -1:
        return "进程被强制终止"
    case 3:
        return "内存不足"
    default:
        return fmt.Sprintf("异常退出（代码 %d）", code)
    }
}
```

```go
    // 1. 创建备份
    backupPath, err := CreateBackup(mcDir)

    os.MkdirAll(filepath.Join(mcDir, "config"), 0755)

    fullSync(mcDir)
    
    // 4. 重新启动
    fmt.Println("")
    fmt.Println("✓ 修复完成！重新启动中...")
    launchPCL2(findPCL2Path())
}
```

## 三、Phase 4：静默守护（托盘模式）

### 3.1 设计

当 MC 成功启动后，更新器不退出，而是：
1. 最小化到系统托盘
2. 静默监听 Minecraft 日志和崩溃报告目录
3. 检测到崩溃时弹出提示

**更新器此时就是一个轻量级的守护进程。**

```go
func enterDaemonMode(mcDir string, pcl2Path string) {
    // 记录启动时间
    startTime := time.Now()
    
    // 显示托盘提示
    showTrayNotification("MC-Starter", "Minecraft 已启动，更新器将在后台守护")
    
    // 主循环：监听崩溃
    ticker := time.NewTicker(5 * time.Second)
    defer ticker.Stop()
    
    for range ticker.C {
        // 检查 PCL2/MC 进程是否还在运行
        if !isProcessRunning("Plain Craft Launcher 2.exe") && 
           !isProcessRunning("javaw.exe") {
            // 都退出了 → 更新器也退出
            log("PCL2 和 MC 均已退出，守护结束")
            showTrayNotification("MC-Starter", "Minecraft 已退出")
            os.Exit(0)
        }
        
        // 检查崩溃报告
        if hasNewCrashReport(mcDir, startTime) {
            crash := getLatestCrash(mcDir)
            showTrayNotification("MC-Starter崩溃了", 
                fmt.Sprintf("Minecraft 发生崩溃:%s", crash.Reason))
            // 弹出修复确认
            handleCrash(crash, mcDir)
            break
        }
        
        // 检查 JVM 崩溃日志
        if hasNewJVMError(mcDir, startTime) {
            crash := CrashInfo{
                IsCrash: true,
                Reason:  "JVM 崩溃（hs_err_pid）",
            }
            showTrayNotification("MC-Starter JVM崩溃", 
                "Java 虚拟机发生崩溃")
            handleCrash(crash, mcDir)
            break
        }
        
        // 检查 MC 日志中的异常（实时检测）
        if hasRecentExceptionInLog(mcDir, 30) {
            // 持续出现异常但不一定崩溃，记录但不打扰用户
            log("MC 日志中检测到异常（非致命）")
        }
    }
}
```

### 3.3 日志监听

```go
func monitorMCLogs(mcDir string, startTime time.Time) {
    logDir := filepath.Join(mcDir, "logs")
    latestLog := filepath.Join(logDir, "latest.log")
    
    if _, err := os.Stat(latestLog); os.IsNotExist(err) {
        return  // 还没生成日志
    }
    
    file, _ := os.Open(latestLog)
    defer file.Close()
    
    // 定位到文件末尾
    file.Seek(0, io.SeekEnd)
    
    reader := bufio.NewReader(file)
    for {
        line, err := reader.ReadString('\n')
        if err != nil {
            time.Sleep(500 * time.Millisecond)
            continue
        }
        
        // 实时检测崩溃关键词
        if strings.Contains(line, "FATAL") || 
           strings.Contains(line, "NullPointerException") ||
           strings.Contains(line, "OutOfMemoryError") ||
           strings.Contains(line, "StackOverflowError") {
            log("[日志警告] %s", strings.TrimSpace(line))
        }
        
        // 检测模组加载错误
        if strings.Contains(line, "Failed to load mod") ||
           strings.Contains(line, "Caused by:") {
            log("[模组错误] %s", strings.TrimSpace(line))
        }
    }
}
```

```go
const crashReportURL = "https://你的服务器.com/api/crash-report"

func uploadCrashReport(crash CrashInfo) {
    if !promptUploadConfirmation() {
        fmt.Println("已取消上传")
        return
    }
    
    // 收集信息
    report := collectCrashReport(crash)
    
    // 上传
    data, _ := json.Marshal(report)
    resp, err := http.Post(crashReportURL, "application/json", 
        bytes.NewReader(data))
    
    if err != nil {
        fmt.Printf("⚠ 上传失败: %v\n", err)
        return
    }
    defer resp.Body.Close()
    
    if resp.StatusCode == 200 {
        // 服务器可能返回建议
        var advice struct {
            Message string `json:"message"`
        }
        json.NewDecoder(resp.Body).Decode(&advice)
        if advice.Message != "" {
            fmt.Printf("💡 服务器建议: %s\n", advice.Message)
        }
    }
}
```
